- 作用：将**有交集**的区间合并，减少区间个数
- 思路：按左端点排序 -> 扫描整个区间、**分类讨论**（可画图理解）左右端点 -> 分别合并

```cpp
// 将所有存在交集的区间合并
void merge(vector<PII> &segs)
{
    vector<PII> ans; // 存储结果
    int st = -2e9, ed = -2e9; // 两个变量分别代表当前结果ans的起止点，初始为无穷
    
    sort(segs.begin(), segs.end());  // sort会优先以左端点排序

    for (auto seg : segs) // 遍历每个区间
        if (ed < seg.first)  // 当前结果ans与原区间reg无交集时
        {
            if (st != -2e9) res.push_back({st, ed}); // 此时若原区间reg不为初始区间，则将其加入答案ans中
            st = seg.first, ed = seg.second; // 并向后移一个区间
        }
        else ed = max(ed, seg.second); // 当前结果ans与原区间reg有交集时，保留较长的一个（最大值）作为答案

    if (st != -2e9) res.push_back({st, ed}); // 将最后一段加入答案（判断条件是为了防止输入为空）

    segs = res; //（保存答案）
}
```
