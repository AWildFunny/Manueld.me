- 作用：一般是**优化**某些问题
## 1. 存储
```cpp
// e[]表示节点的值，l[]表示节点的左指针，r[]表示节点的右指针，idx表示当前用到了哪个节点
int e[N], // 节点的值
int l[N], // 左指针
int r[N], // 右指针
int idx; // 索引
```
## 2. 初始化
```cpp
// 初始化思路：链接前两个点
void init()
{
    //0是左端点，1是右端点
    r[0] = 1, l[1] = 0;  // 索引为0、1的头两个元素，左右指针相互连接
    idx = 2; // 初始化索引为第三个元素
} 
```
## 3. 插入
```cpp
// 在节点a的右边插入一个数x
void insert(int a, int x)
{ // idx代表当前操作的元素索引（未插入链表的元素）
    e[idx] = x; // 赋值
    l[idx] = a, r[idx] = r[a]; // 插入元素的指针赋地址
    l[r[a]] = idx, r[a] = idx; // 左右元素的指针赋地址（注意先右后左）
    idx ++; // 当前索引递增
}
```
## 4. 删除
```cpp
// 删除节点a
void remove(int a)
{
    l[r[a]] = l[a]; // 先改变右元素的指针
    r[l[a]] = r[a]; // 再改变左元素的指针
}
```
